GameEngineLib.Factory = function(instance, private)
{
	instance = instance || {};
	private = private || {};
	
	if(GameEngineLib.DEBUG)
	{
		GameEngineLib.addDebugInfo("Factory", instance, private);
	}
	
	private.userCreateFunc = function(inName){};
		
	instance.setCreate = function(inCreateFunc)
	{
		private.userCreateFunc = inCreateFunc;
	}
	
	instance.create = function()//params var instead to passthru?
	{
		return private.userCreateFunc(arguments);
	}
	
	return instance;
}
